JS Guide JS Tips

Variables & Types

Declaring Variables

const name  = "Myat";     // never changes
let   score = 0;          // will change
// var — avoid completely

Data Types

const str  = "hello";       // String
const num  = 42;            // Number
const bool = true;          // Boolean
const arr  = [1, 2, 3];     // Array
const obj  = { a: 1 };      // Object
const nil  = null;          // Null
let   x;                    // Undefined

Type Checking

typeof "hello"    // "string"
typeof 42         // "number"
typeof true       // "boolean"
typeof undefined  // "undefined"
typeof null       // "object" (JS bug!)
typeof []         // "object"
Array.isArray([]) // true

Type Conversion

Number("42")        // 42
Number("abc")       // NaN
String(42)          // "42"
Boolean(0)          // false
Boolean("hello")    // true
parseInt("42px")    // 42
parseFloat("3.14x") // 3.14

Operators

Comparison

5 === 5      // true  (use this)
5 !== 3      // true
5 == "5"     // true  (avoid!)
5 > 3        // true
5 >= 5       // true
5 < 10       // true

Logical

true && true    // true  (AND)
true || false   // true  (OR)
!true           // false (NOT)
null ?? "default"  // "default" (nullish)
0    ?? "default"  // 0 (0 is valid!)

Arithmetic

10 + 3  // 13
10 - 3  // 7
10 * 3  // 30
10 / 3  // 3.33
10 % 3  // 1  (remainder)
10 ** 2 // 100 (power)
x++     // increment
x--     // decrement

Shorthand Assignment

x += 5   // x = x + 5
x -= 5   // x = x - 5
x *= 2   // x = x * 2
x /= 2   // x = x / 2
x **= 2  // x = x * x
x ??= 10 // x = x ?? 10

Functions

Function Types

// Declaration
function greet(name) {
  return `Hello, ${name}`;
}

// Expression
const greet = function(name) {
  return `Hello, ${name}`;
};

// Arrow (use this most)
const greet = (name) => `Hello, ${name}`;

Default & Rest Params

// Default parameter
const greet = (name = "Guest") => name;
greet();       // "Guest"
greet("Myat"); // "Myat"

// Rest — collect into array
const sum = (...nums) =>
  nums.reduce((a, b) => a + b, 0);
sum(1, 2, 3, 4); // 10

Ternary & Short-circuit

// Ternary
const status = age >= 18 ? "Adult" : "Minor";

// &&  — run if truthy
user && greet(user);

// ||  — fallback if falsy
const name = user.name || "Guest";

// ??  — fallback if null/undefined
const count = data.count ?? 0;

Destructured Parameters

// Destructure object param
const display = ({ name, age, city = "Unknown" }) => {
  console.log(`${name}, ${age}, ${city}`);
};

display({ name: "Myat", age: 19 });

// Destructure array param
const first = ([a, b]) => a;
first([10, 20]); // 10

Array Methods

Transform

// map — transform each item
[1,2,3].map(n => n * 2)
// [2, 4, 6]

// filter — keep matching items
[1,2,3,4].filter(n => n > 2)
// [3, 4]

// reduce — collapse to one value
[1,2,3,4].reduce((sum, n) => sum + n, 0)
// 10

Find & Check

// find — first match (or undefined)
[3,7,2].find(n => n > 5)     // 7

// findIndex — index of first match
[3,7,2].findIndex(n => n > 5) // 1

// some — any match?
[1,2,3].some(n => n > 2)     // true

// every — all match?
[1,2,3].every(n => n > 0)    // true

// includes — value in array?
[1,2,3].includes(2)          // true

Add / Remove

arr.push(item)    // add to end
arr.pop()         // remove from end
arr.unshift(item) // add to start
arr.shift()       // remove from start

// splice(start, deleteCount, ...items)
arr.splice(2, 1)        // remove 1 at index 2
arr.splice(1, 0, "new") // insert at index 1

// slice — copy portion (non-destructive)
arr.slice(1, 3)  // items at index 1 and 2

Useful Patterns

// Sort numbers
arr.sort((a, b) => a - b); // ascending
arr.sort((a, b) => b - a); // descending

// Copy array
const copy = [...arr];

// Remove duplicates
const unique = [...new Set(arr)];

// Flatten nested arrays
[[1,2],[3,4]].flat(); // [1,2,3,4]

// Join to string
["a","b","c"].join(", "); // "a, b, c"

Objects & Destructuring

Object Basics

const user = { name: "Myat", age: 19 };

user.name          // dot notation
user["name"]       // bracket notation
user.email = "x"   // add property
delete user.age    // remove property

// Shorthand (same name)
const name = "Myat";
const obj = { name }; // { name: "Myat" }

Destructuring

const { name, age } = user;
// Rename: const { name: fullName } = user;
// Default: const { role = "user" } = user;

// Array destructuring
const [first, , third] = [1, 2, 3];

// Nested
const { address: { city } } = user;

// In function params
const fn = ({ name, age }) => name;

Spread & Merge

// Copy
const copy = { ...obj };

// Merge (later wins)
const merged = { ...defaults, ...overrides };

// Update one field
const updated = { ...user, age: 20 };

// Rest — collect remaining
const { name, ...rest } = user;
// rest = everything except name

Object Methods

Object.keys(obj)    // ["name", "age"]
Object.values(obj)  // ["Myat", 19]
Object.entries(obj) // [["name","Myat"],["age",19]]

// Loop over entries
Object.entries(obj).forEach(([key, val]) => {
  console.log(key, val);
});

// Check if key exists
"name" in obj       // true
obj.hasOwnProperty("name") // true

DOM Manipulation

Selecting Elements

document.getElementById("id")
document.querySelector(".class")    // first match
document.querySelector("#id")
document.querySelector("tag")
document.querySelectorAll(".class") // all matches (NodeList)

// Convert NodeList to Array
Array.from(document.querySelectorAll(".item"))
[...document.querySelectorAll(".item")]

Reading & Writing

el.textContent         // read/write text (safe)
el.innerHTML           // read/write HTML
el.value               // input field value
el.getAttribute("src") // read attribute
el.setAttribute("src", "img.png")
el.removeAttribute("disabled")
el.dataset.id          // read data-id attr

Classes & Styles

el.classList.add("active")
el.classList.remove("hidden")
el.classList.toggle("dark")
el.classList.contains("active") // true/false
el.classList.replace("old", "new")

// Inline style (use classList instead when possible)
el.style.color = "red";
el.style.display = "none";

Creating & Inserting

const el = document.createElement("div");
el.classList.add("card");
el.textContent = "Hello";

parent.appendChild(el)   // add to end
parent.prepend(el)        // add to start
el.remove()              // remove element

// Insert HTML string
parent.insertAdjacentHTML("beforeend", `
  <div class="card">${data}</div>
`);

Events

Adding Listeners

el.addEventListener("click", handler);
el.addEventListener("input", handler);
el.addEventListener("submit", handler);
el.addEventListener("keydown", handler);
el.addEventListener("mouseover", handler);
el.removeEventListener("click", handler);

// Once — auto-removes after first fire
el.addEventListener("click", fn, { once: true });

Event Object

el.addEventListener("click", (e) => {
  e.target           // element clicked
  e.currentTarget    // element with listener
  e.preventDefault() // stop default behavior
  e.stopPropagation()// stop bubbling up

  e.target.value     // input value
  e.key              // "Enter", "Escape", etc.
  e.clientX          // mouse X position
});

Event Delegation

// One listener on parent — handles all children
list.addEventListener("click", (e) => {
  if (e.target.matches(".delete-btn")) {
    e.target.closest("li").remove();
  }
  if (e.target.matches(".edit-btn")) {
    const id = e.target.dataset.id;
    editItem(id);
  }
});

Form Events

form.addEventListener("submit", (e) => {
  e.preventDefault(); // stop page reload!

  const data = {
    name:  form.querySelector("#name").value.trim(),
    email: form.querySelector("#email").value.trim()
  };

  if (!data.name) return; // validate
  processData(data);
  form.reset();
});

Async / Fetch

Basic Fetch

const getData = async () => {
  try {
    const res  = await fetch(url);
    if (!res.ok) throw new Error(res.status);
    const data = await res.json();
    return data;
  } catch (err) {
    console.error(err);
    return null;
  }
};

POST Request

const postData = async (url, body) => {
  const res = await fetch(url, {
    method:  "POST",
    headers: { "Content-Type": "application/json" },
    body:    JSON.stringify(body)
  });
  return res.json();
};

Promise Patterns

// Parallel — fastest
const [a, b] = await Promise.all([
  fetch(url1).then(r => r.json()),
  fetch(url2).then(r => r.json())
]);

// First to resolve
const first = await Promise.race([fn1(), fn2()]);

// All, even if some fail
const results = await Promise.allSettled([fn1(), fn2()]);
results.forEach(r => r.status === "fulfilled" ? use(r.value) : log(r.reason));

Async Patterns

// Sequential (each waits for prev)
for (const id of ids) {
  const data = await fetch(id);
}

// Parallel (all at once)
await Promise.all(ids.map(id => fetch(id)));

// Debounce async
const search = debounce(async (q) => {
  const results = await fetchSearch(q);
  render(results);
}, 400);

Optional Chaining & Nullish

obj?.prop           // undefined if obj is null
obj?.method?.()     // only calls if exists
arr?.[0]            // optional array access

value ?? "default"  // fallback for null/undefined
value ||  "default" // fallback for any falsy
value &&  doSomething() // run if truthy
