Selectors

Basic Selectors

*            /* all elements */
div          /* by tag */
.class       /* by class */
#id          /* by id */
div.class    /* tag + class */
div, p       /* multiple */

Combinators

div p        /* descendant */
div > p      /* direct child */
div + p      /* adjacent sibling */
div ~ p      /* all siblings */

Pseudo-classes

:hover       :focus
:active      :visited
:first-child :last-child
:nth-child(2)
:nth-child(odd)
:nth-child(even)
:not(.class)
:is(h1, h2, h3)

Pseudo-elements

::before     /* insert before */
::after      /* insert after */
::placeholder
::selection  /* selected text */
::first-line
::first-letter

/* Always need content: */
.el::before { content: ""; }

Attribute Selectors

[type="text"]   /* exact match */
[class*="btn"]  /* contains */
[href^="https"] /* starts with */
[src$=".png"]   /* ends with */
[data-id]       /* has attribute */

Specificity Order

/* Lowest to Highest */
* { }              /* 0,0,0 */
p { }              /* 0,0,1 */
.class { }         /* 0,1,0 */
#id { }            /* 1,0,0 */
style=""           /* inline */
!important         /* nuclear */

/* !important — avoid, use specificity */

Box Model & Spacing

Box Model

/* Always use border-box */
* { box-sizing: border-box; }

/* Layers (outside → inside) */
margin → border → padding → content

margin: 10px 20px;      /* top/bot left/right */
margin: 10px 20px 5px;  /* top lr bottom */
margin: 10px 20px 5px 0;/* top r bot left */
margin: auto;           /* center horizontally */

Width & Height

width: 100%;
width: 300px;
width: 50vw;
min-width: 200px;
max-width: 1200px;

height: 100vh;
height: auto;    /* shrink to content */
min-height: 100vh;

/* Responsive width */
width: clamp(300px, 50%, 800px);

Display

display: block;
display: inline;
display: inline-block;
display: flex;
display: grid;
display: none;     /* hide + remove space */

/* Hide but keep space */
visibility: hidden;
opacity: 0;

Overflow

overflow: visible;  /* default */
overflow: hidden;   /* clip content */
overflow: scroll;   /* always scroll */
overflow: auto;     /* scroll if needed */

overflow-x: hidden;  /* horizontal */
overflow-y: auto;    /* vertical */

/* Truncate text */
white-space: nowrap;
overflow: hidden;
text-overflow: ellipsis;

Flexbox

Container Properties

display: flex;
flex-direction: row;       /* default */
flex-direction: column;
flex-direction: row-reverse;

flex-wrap: nowrap;         /* default */
flex-wrap: wrap;
flex-wrap: wrap-reverse;

/* Shorthand */
flex-flow: row wrap;

Alignment

/* Main axis (flex-direction) */
justify-content: flex-start;  /* default */
justify-content: flex-end;
justify-content: center;
justify-content: space-between;
justify-content: space-around;
justify-content: space-evenly;

/* Cross axis */
align-items: stretch;  /* default */
align-items: flex-start;
align-items: flex-end;
align-items: center;
align-items: baseline;

Gap & Multi-line

gap: 16px;           /* row + column */
gap: 10px 20px;      /* row column */
row-gap: 10px;
column-gap: 20px;

/* Multi-line (when wrapping) */
align-content: flex-start;
align-content: center;
align-content: space-between;

Item Properties

/* Grow — take available space */
flex-grow: 0;    /* default — don't grow */
flex-grow: 1;    /* grow and fill space */

/* Shrink — below basis */
flex-shrink: 1;  /* default — can shrink */
flex-shrink: 0;  /* don't shrink */

/* Base size */
flex-basis: auto;
flex-basis: 200px;

/* Shorthand */
flex: 1;         /* grow:1 shrink:1 basis:0 */
flex: 0 0 200px; /* fixed 200px */

/* Override alignment for one item */
align-self: center;

CSS Grid

Defining Columns & Rows

display: grid;

grid-template-columns: 200px 1fr 1fr;
grid-template-columns: repeat(3, 1fr);
grid-template-columns: repeat(auto-fit, minmax(250px, 1fr));

grid-template-rows: auto 1fr auto;
grid-auto-rows: minmax(100px, auto);

gap: 20px;
column-gap: 20px;
row-gap: 10px;

Template Areas

grid-template-areas:
  "header header"
  "sidebar main"
  "footer footer";

grid-template-columns: 240px 1fr;
grid-template-rows: 60px 1fr 50px;

/* Assign areas */
header  { grid-area: header; }
.sidebar{ grid-area: sidebar; }
main    { grid-area: main; }
footer  { grid-area: footer; }

Placing Items

/* Column span */
grid-column: 1 / 3;      /* col 1 to 3 */
grid-column: 1 / -1;     /* full width */
grid-column: span 2;     /* span 2 cols */

/* Row span */
grid-row: 1 / 3;
grid-row: span 2;

/* Place shorthand */
grid-area: 1 / 1 / 3 / 3;
/* row-start / col-start / row-end / col-end */

Alignment in Grid

/* Align all items */
justify-items: start | center | end | stretch;
align-items:   start | center | end | stretch;

/* Align one item */
justify-self: center;
align-self: end;

/* Align the whole grid */
justify-content: center;
align-content: center;

Position

Position Values

position: static;    /* default — in flow */
position: relative;  /* offset from self */
position: absolute;  /* relative to nearest
                        positioned ancestor */
position: fixed;     /* relative to viewport */
position: sticky;    /* relative until threshold */

/* Offset (top/right/bottom/left) */
top: 0; right: 0; bottom: 0; left: 0;

Common Patterns

/* Center with absolute */
.parent { position: relative; }
.child  {
  position: absolute;
  top: 50%; left: 50%;
  transform: translate(-50%, -50%);
}

/* Sticky header */
header {
  position: sticky;
  top: 0;
  z-index: 100;
}

/* Fixed overlay */
.modal {
  position: fixed;
  inset: 0;          /* top:0 right:0 bottom:0 left:0 */
  z-index: 1000;
}

z-index

/* Only works on positioned elements */
z-index: auto;     /* default */
z-index: 1;        /* above 0 */
z-index: -1;       /* below */
z-index: 9999;     /* top layer */

/* Recommended scale */
--z-base:    1;
--z-overlay: 10;
--z-modal:   100;
--z-toast:   1000;
--z-tooltip: 9999;

Stacking Context

/* Creates a new stacking context */
position: relative + z-index (not auto)
position: fixed or sticky
opacity: less than 1
transform: any value
filter: any value
isolation: isolate  /* explicit */

/* Fix z-index not working */
.parent { isolation: isolate; }
/* Each parent has its own z-index stack */

Responsive Text

/* Scale heading with viewport */

/* Fluid line height */
line-height: clamp(1.4, 1.5 + 0.3vw, 1.8);

Transitions & Animations

Transitions

/* property duration timing delay */
transition: all 0.3s ease;
transition: opacity 0.2s ease, transform 0.3s ease;

/* Timing functions */
ease          /* slow → fast → slow */
ease-in       /* slow start */
ease-out      /* slow end */
ease-in-out
linear
cubic-bezier(0.4, 0, 0.2, 1)  /* material */

Transforms

transform: translateX(20px);
transform: translateY(-50%);
transform: translate(-50%, -50%); /* center trick */

transform: scale(1.05);
transform: scaleX(0.5);

transform: rotate(45deg);
transform: rotateY(180deg);

transform: skew(10deg);

/* Chain multiple */
transform: translateY(-4px) scale(1.02);

@keyframes

@keyframes fadeIn {
  from { opacity: 0; transform: translateY(10px); }
  to   { opacity: 1; transform: translateY(0); }
}

@keyframes spin {
  from { transform: rotate(0deg); }
  to   { transform: rotate(360deg); }
}

/* Apply: name duration timing iteration */
animation: fadeIn 0.4s ease forwards;
animation: spin 1s linear infinite;

Animation Properties

animation-name: fadeIn;
animation-duration: 0.4s;
animation-timing-function: ease;
animation-delay: 0.2s;
animation-iteration-count: 1 | infinite;
animation-direction: normal | reverse | alternate;
animation-fill-mode: forwards | backwards | both;
animation-play-state: running | paused;

CSS Variables & Responsive

CSS Variables

:root {
  --color-primary: #e94560;
  --color-bg:      #0f0f1a;
  --font-size-base: 1rem;
  --spacing-md:    16px;
  --radius:        8px;
}

/* Usage */
.btn {
  background: var(--color-primary);
  padding: var(--spacing-md);
  border-radius: var(--radius);
}

/* Fallback */
color: var(--color-text, #fff);

Media Queries

/* Mobile first (recommended) */
/* Base styles = mobile */

@media (min-width: 640px)  { /* sm  */ }
@media (min-width: 768px)  { /* md  */ }
@media (min-width: 1024px) { /* lg  */ }
@media (min-width: 1280px) { /* xl  */ }

/* Dark mode */
@media (prefers-color-scheme: dark) { }

/* Reduced motion */
@media (prefers-reduced-motion: reduce) {
  * { animation: none !important; }
}

CSS Units

px   /* absolute pixels */
rem  /* relative to root font-size (16px) */
em   /* relative to parent font-size */
%    /* relative to parent */
vw   /* viewport width (100vw = full) */
vh   /* viewport height */
vmin /* smaller of vw/vh */
vmax /* larger of vw/vh */
ch   /* width of "0" character */
fr   /* grid fraction unit */

Useful Functions

/* Clamp — responsive sizing */
font-size: clamp(1rem, 2.5vw, 2rem);
width: clamp(300px, 50%, 800px);

/* min/max */
width: min(100%, 600px);
padding: max(20px, 5%);

/* calc */
width: calc(100% - 240px);
height: calc(100vh - 60px);
padding: calc(var(--spacing) * 2);
